# Test9. a ^ b mod c

PROCEDURE power(a,b,c,d) IS
    pot,wyk,o
IN
    d:=1;
    wyk:=b;
    pot:=a%c;
    WHILE wyk>0 DO
	o:=wyk%2;
	IF o=1 THEN
	    d:=d*pot;
	    d:=d%c;
	ENDIF
	wyk:=wyk/2;
	pot:=pot*pot;
	pot:=pot%c;
    ENDWHILE
END

PROGRAM IS
   a,b,c,d
IN
   a:=1234567890;
   b:=1234567890987654321;
   c:=987654321;
   power(a,b,c,d);
   WRITE d;
END

# Test15. GCD

PROCEDURE gcd(a,b,c) IS
  x,y
IN
  x:=a;
  y:=b;
  WHILE y>0 DO
    IF x>=y THEN 
      x:=x-y;
    ELSE 
      x:=x+y;
      y:=x-y;
      x:=x-y;    
    ENDIF
  ENDWHILE
  c:=x;
END

PROGRAM IS
  a,b,c,d,x,y,z
IN
  a:=80;
  b:=100;
  c:=150;
  d:=10000690;
  gcd(a,b,x);
  gcd(c,d,y);
  gcd(x,y,z);
  WRITE z;
END
